% 学术内容检测模块核心领域类图
\begin{tikzpicture}[
  class/.style={
    draw,
    rectangle split,
    rectangle split parts=3,
    rectangle split part align={center,left,left},
    align=left,
    text width=2.75cm,
    font=\scriptsize,
    inner sep=2.5pt
  },
  rel/.style={-{Stealth[length=1.6mm]},semithick},
  dep/.style={-{Stealth[length=1.6mm]},dashed,semithick}
]
  \node[class] (resource) at (0,0) {
    \textbf{学术资源}
    \nodepart{second}
    - 类型, 原始文件
    \nodepart{third}
    + 解析内容()\\
    + 标准化输入()
  };

  \node[class] (image) at (4.2,1.6) {
    \textbf{学术图像}
    \nodepart{second}
    - 来源页码, 编号
    \nodepart{third}
    + 提取图像()\\
    + 生成标注()
  };

  \node[class] (paper) at (4.2,0) {
    \textbf{论文文档}
    \nodepart{second}
    - 标题, 段落
    \nodepart{third}
    + 分段解析()\\
    + 分析AIGC概率()
  };

  \node[class] (review) at (4.2,-1.6) {
    \textbf{同行评审Review}
    \nodepart{second}
    - 文本, 模板特征
    \nodepart{third}
    + 检测虚假评审()\\
    + 分析模板倾向()
  };

  \node[class] (task) at (8.4,0) {
    \textbf{内容检测任务}
    \nodepart{second}
    - 检测类型, 状态
    \nodepart{third}
    + 启动检测()\\
    + 记录过程()
  };

  \node[class] (result) at (12.6,0.8) {
    \textbf{检测结果}
    \nodepart{second}
    - 风险, 置信度
    \nodepart{third}
    + 生成子结论()\\
    + 输出解释数据()
  };

  \node[class] (report) at (12.6,-0.8) {
    \textbf{综合鉴伪报告}
    \nodepart{second}
    - 综合结论, 格式
    \nodepart{third}
    + 融合结果()\\
    + 导出报告()
  };

  \draw[rel] (resource) -- (image);
  \draw[rel] (resource) -- (paper);
  \draw[rel] (resource) -- (review);
  \draw[rel] (image) -- (task);
  \draw[rel] (paper) -- (task);
  \draw[rel] (review) -- (task);
  \draw[rel] (task) -- (result);
  \draw[dep] (result) -- (report);
\end{tikzpicture}
