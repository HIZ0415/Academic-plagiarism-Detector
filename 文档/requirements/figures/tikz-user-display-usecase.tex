% 用户展示模块服务协作类图
\begin{tikzpicture}[
  class/.style={
    draw,
    rectangle split,
    rectangle split parts=3,
    rectangle split part align={center,left,left},
    align=left,
    text width=2.65cm,
    font=\scriptsize,
    inner sep=2.5pt
  },
  rel/.style={-{Stealth[length=1.6mm]},semithick},
  dep/.style={-{Stealth[length=1.6mm]},dashed,semithick}
]
  \node[class] (controller) at (0,0) {
    \textbf{展示控制器}
    \nodepart{second}
    - 用户上下文
    \nodepart{third}
    + 获取个人中心()\\
    + 获取任务详情()
  };

  \node[class] (service) at (4.2,0) {
    \textbf{展示聚合服务}
    \nodepart{second}
    - 查询条件
    \nodepart{third}
    + 聚合首页数据()\\
    + 组装详情视图()
  };

  \node[class] (recordSvc) at (8.4,2.2) {
    \textbf{记录查询服务}
    \nodepart{second}
    - 时间范围, 类型
    \nodepart{third}
    + 查询检测记录()\\
    + 查询审核记录()
  };

  \node[class] (msgSvc) at (8.4,0) {
    \textbf{消息服务}
    \nodepart{second}
    - 消息类型, 状态
    \nodepart{third}
    + 查询反馈()\\
    + 更新已读()
  };

  \node[class] (reportSvc) at (8.4,-2.2) {
    \textbf{报告服务}
    \nodepart{second}
    - 报告ID, 令牌
    \nodepart{third}
    + 校验权限()\\
    + 创建下载链接()
  };

  \node[class] (repo) at (12.6,0) {
    \textbf{展示数据仓储}
    \nodepart{second}
    - 用户/任务/消息表
    \nodepart{third}
    + 按用户查询()\\
    + 按条件筛选()
  };

  \draw[rel] (controller) -- (service);
  \draw[rel] (service) -- (recordSvc);
  \draw[rel] (service) -- (msgSvc);
  \draw[rel] (service) -- (reportSvc);
  \draw[dep] (recordSvc) -- (repo);
  \draw[dep] (msgSvc) -- (repo);
  \draw[dep] (reportSvc) -- (repo);
\end{tikzpicture}
