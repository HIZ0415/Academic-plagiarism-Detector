% 用户上传模块服务协作类图
\begin{tikzpicture}[
  class/.style={
    draw,
    rectangle split,
    rectangle split parts=3,
    rectangle split part align={center,left,left},
    align=left,
    text width=2.75cm,
    font=\scriptsize,
    inner sep=2.5pt
  },
  rel/.style={-{Stealth[length=1.6mm]},semithick},
  dep/.style={-{Stealth[length=1.6mm]},dashed,semithick}
]
  \node[class] (controller) at (0,0) {
    \textbf{上传控制器}
    \nodepart{second}
    - 上传请求
    \nodepart{third}
    + 接收文件()\\
    + 查询状态()
  };

  \node[class] (validator) at (4.2,1.5) {
    \textbf{文件校验器}
    \nodepart{second}
    - 格式, 大小上限
    \nodepart{third}
    + 校验格式()\\
    + 校验完整性()
  };

  \node[class] (storage) at (4.2,-1.5) {
    \textbf{文件存储服务}
    \nodepart{second}
    - 存储路径, 摘要
    \nodepart{third}
    + 保存文件()\\
    + 生成地址()
  };

  \node[class] (taskSvc) at (8.4,0) {
    \textbf{任务创建服务}
    \nodepart{second}
    - 任务类型, 入队策略
    \nodepart{third}
    + 创建检测任务()\\
    + 创建批量任务()
  };

  \node[class] (scheduler) at (12.6,1.5) {
    \textbf{检测调度器}
    \nodepart{second}
    - 队列, 并发阈值
    \nodepart{third}
    + 投递作业()\\
    + 控制并发()
  };

  \node[class] (status) at (12.6,-1.5) {
    \textbf{状态与审核服务}
    \nodepart{second}
    - 进度, 审核请求
    \nodepart{third}
    + 推送状态()\\
    + 创建审核请求()
  };

  \draw[rel] (controller) -- (validator);
  \draw[rel] (controller) -- (storage);
  \draw[rel] (validator) -- (taskSvc);
  \draw[rel] (storage) -- (taskSvc);
  \draw[rel] (taskSvc) -- (scheduler);
  \draw[dep] (scheduler) -- (status);
  \draw[dep] (taskSvc) -- (status);
\end{tikzpicture}
