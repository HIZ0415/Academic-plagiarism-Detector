% 用户审核模块服务协作类图
\begin{tikzpicture}[
  class/.style={
    draw,
    rectangle split,
    rectangle split parts=3,
    rectangle split part align={center,left,left},
    align=left,
    text width=2.65cm,
    font=\scriptsize,
    inner sep=2.5pt
  },
  rel/.style={-{Stealth[length=1.6mm]},semithick},
  dep/.style={-{Stealth[length=1.6mm]},dashed,semithick}
]
  \node[class] (controller) at (0,0) {
    \textbf{审核控制器}
    \nodepart{second}
    - 用户上下文
    \nodepart{third}
    + 查询任务池()\\
    + 提交审核操作()
  };

  \node[class] (taskSvc) at (4.2,2.2) {
    \textbf{审核任务服务}
    \nodepart{second}
    - 状态, 筛选规则
    \nodepart{third}
    + 查询待审任务()\\
    + 更新任务进度()
  };

  \node[class] (opinionSvc) at (4.2,0) {
    \textbf{意见服务}
    \nodepart{second}
    - 结论, 理由
    \nodepart{third}
    + 保存意见()\\
    + 统计结果()
  };

  \node[class] (commentSvc) at (4.2,-2.2) {
    \textbf{评论举报服务}
    \nodepart{second}
    - 评论, 举报信息
    \nodepart{third}
    + 发布评论()\\
    + 提交举报()
  };

  \node[class] (notify) at (8.4,1.25) {
    \textbf{通知服务}
    \nodepart{second}
    - 接收用户, 类型
    \nodepart{third}
    + 发送进度通知()\\
    + 发送结果通知()
  };

  \node[class] (repo) at (8.4,-1.25) {
    \textbf{审核数据仓储}
    \nodepart{second}
    - 任务/意见/评论表
    \nodepart{third}
    + 保存审核数据()\\
    + 查询审核数据()
  };

  \draw[rel] (controller) -- (taskSvc);
  \draw[rel] (controller) -- (opinionSvc);
  \draw[rel] (controller) -- (commentSvc);
  \draw[rel] (opinionSvc) -- (notify);
  \draw[dep] (taskSvc) -- (repo);
  \draw[dep] (opinionSvc) -- (repo);
  \draw[dep] (commentSvc) -- (repo);
\end{tikzpicture}
