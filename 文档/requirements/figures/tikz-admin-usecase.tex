% 管理维护模块服务协作类图
\begin{tikzpicture}[
  class/.style={
    draw,
    rectangle split,
    rectangle split parts=3,
    rectangle split part align={center,left,left},
    align=left,
    text width=2.55cm,
    font=\scriptsize,
    inner sep=2pt
  },
  rel/.style={-{Stealth[length=1.6mm]},semithick},
  dep/.style={-{Stealth[length=1.6mm]},dashed,semithick}
]
  \node[class] (controller) at (0,0) {
    \textbf{管理端控制器}
    \nodepart{second}
    - 管理员上下文
    \nodepart{third}
    + 分发操作()\\
    + 校验权限()
  };

  \node[class] (entitySvc) at (4.2,2.4) {
    \textbf{实体管理服务}
    \nodepart{second}
    - 实体类型, 标签
    \nodepart{third}
    + 添加实体()\\
    + 维护实体()
  };

  \node[class] (statSvc) at (4.2,0.8) {
    \textbf{统计服务}
    \nodepart{second}
    - 维度, 时间范围
    \nodepart{third}
    + 生成统计()\\
    + 导出报表()
  };

  \node[class] (auditSvc) at (4.2,-0.8) {
    \textbf{日志审计服务}
    \nodepart{second}
    - 日志类型, 条件
    \nodepart{third}
    + 记录日志()\\
    + 查询审计()
  };

  \node[class] (modelSvc) at (4.2,-2.4) {
    \textbf{模型配置服务}
    \nodepart{second}
    - 参数, 启用策略
    \nodepart{third}
    + 更新配置()\\
    + 验证模型()
  };

  \node[class] (moderation) at (8.4,1.35) {
    \textbf{举报审核服务}
    \nodepart{second}
    - 举报信息, 策略
    \nodepart{third}
    + 审核举报()\\
    + 发送结果()
  };

  \node[class] (repo) at (8.4,-1.35) {
    \textbf{管理数据仓储}
    \nodepart{second}
    - 用户/资源/日志表
    \nodepart{third}
    + 保存管理数据()\\
    + 查询管理数据()
  };

  \draw[rel] (controller) -- (entitySvc);
  \draw[rel] (controller) -- (statSvc);
  \draw[rel] (controller) -- (auditSvc);
  \draw[rel] (controller) -- (modelSvc);
  \draw[rel] (auditSvc.east) -- ++(0.6,0) |- (moderation.west);
  \draw[dep] (entitySvc) -- (repo);
  \draw[dep] (statSvc) -- (repo);
  \draw[dep] (auditSvc) -- (repo);
  \draw[dep] (modelSvc) -- (repo);
  \draw[dep] (moderation) -- (repo);
\end{tikzpicture}
