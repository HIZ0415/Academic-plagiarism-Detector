% 用户审核模块核心领域类图
\begin{tikzpicture}[
  class/.style={
    draw,
    rectangle split,
    rectangle split parts=3,
    rectangle split part align={center,left,left},
    align=left,
    text width=2.65cm,
    font=\scriptsize,
    inner sep=2.5pt
  },
  rel/.style={-{Stealth[length=1.6mm]},semithick},
  dep/.style={-{Stealth[length=1.6mm]},dashed,semithick}
]
  \node[class] (user) at (0,0) {
    \textbf{审核用户}
    \nodepart{second}
    - 用户ID, 信誉
    \nodepart{third}
    + 参与审核()\\
    + 举报内容()
  };

  \node[class] (task) at (4.2,0) {
    \textbf{人工审核任务}
    \nodepart{second}
    - 类型, 进度
    \nodepart{third}
    + 发布任务()\\
    + 汇总结论()
  };

  \node[class] (opinion) at (8.4,2.2) {
    \textbf{审核意见}
    \nodepart{second}
    - 结论, 理由
    \nodepart{third}
    + 提交意见()\\
    + 修改意见()
  };

  \node[class] (comment) at (8.4,0) {
    \textbf{评论互动}
    \nodepart{second}
    - 内容, 点赞数
    \nodepart{third}
    + 发布评论()\\
    + 点赞()
  };

  \node[class] (report) at (8.4,-2.2) {
    \textbf{举报记录}
    \nodepart{second}
    - 类型, 处理状态
    \nodepart{third}
    + 提交举报()\\
    + 进入管理审核()
  };

  \node[class] (summary) at (12.6,0) {
    \textbf{人工审核结果}
    \nodepart{second}
    - 最终结论, 风险
    \nodepart{third}
    + 统计意见()\\
    + 生成通知()
  };

  \draw[rel] (user) -- (task);
  \draw[rel] (task) -- (opinion);
  \draw[rel] (task) -- (comment);
  \draw[rel] (task) -- (report);
  \draw[rel] (opinion) -- (summary);
  \draw[dep] (comment) -- (summary);
\end{tikzpicture}
