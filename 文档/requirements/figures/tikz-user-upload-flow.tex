% 用户上传模块核心领域类图
\begin{tikzpicture}[
  class/.style={
    draw,
    rectangle split,
    rectangle split parts=3,
    rectangle split part align={center,left,left},
    align=left,
    text width=2.65cm,
    font=\scriptsize,
    inner sep=2.5pt
  },
  rel/.style={-{Stealth[length=1.6mm]},semithick},
  dep/.style={-{Stealth[length=1.6mm]},dashed,semithick}
]
  \node[class] (user) at (0,0) {
    \textbf{用户}
    \nodepart{second}
    - 用户ID, 登录状态
    \nodepart{third}
    + 上传文件()\\
    + 申请审核()
  };

  \node[class] (file) at (4.2,1.4) {
    \textbf{上传文件}
    \nodepart{second}
    - 类型, 大小
    \nodepart{third}
    + 校验格式()\\
    + 保存文件()
  };

  \node[class] (batch) at (4.2,-1.4) {
    \textbf{批量上传包}
    \nodepart{second}
    - 文件数量, 状态
    \nodepart{third}
    + 拆分文件()\\
    + 批量校验()
  };

  \node[class] (task) at (8.4,0) {
    \textbf{检测任务}
    \nodepart{second}

    \nodepart{third}
    + 创建任务()\\
    + 更新状态()
  };

  \node[class] (job) at (12.6,1.4) {
    \textbf{AI检测作业}
    \nodepart{second}
    - 模型类型, 进度
    \nodepart{third}
    + 执行检测()\\
    + 同步进度()
  };

  \node[class] (result) at (12.6,-1.4) {
    \textbf{检测结果}
    \nodepart{second}
    - 风险等级, 报告
    \nodepart{third}
    + 生成报告()\\
    + 提交复核()
  };

  \draw[rel] (user) -- (file);
  \draw[rel] (user) -- (batch);
  \draw[rel] (file) -- (task);
  \draw[rel] (batch) -- (task);
  \draw[rel] (task) -- (job);
  \draw[rel] (job) -- (result);
  \draw[dep] (task) -- (result);
\end{tikzpicture}
